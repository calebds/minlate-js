JavaScript

Function to render template on click:

var renderTemplate1 = function() {      
  minlate(
    id : "template1",
    values : {
      first: find("template1-first").value,
      last: find("template1-last").value,
      email: find("template1-email").value,
      phone: find("template1-phone").value
    },
    inPlace : true
  });
};

where 'find' is an alias for 'document.getElementById'.
